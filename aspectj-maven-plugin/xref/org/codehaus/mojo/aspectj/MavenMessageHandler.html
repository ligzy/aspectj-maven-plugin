27  import org.apache.maven.plugin.logging.Log;
28  import org.aspectj.bridge.IMessage;
29  import org.aspectj.bridge.ISourceLocation;
30  import org.aspectj.bridge.MessageHandler;
31  
32  import java.util.ArrayList;
33  import java.util.List;
34  
35  /**
36   * MessageHandler implementation which uses the standard Maven Log to emit
37   * messages from the AJC process. For warnings and error messages from the AJC,
38   * the message detail (containing information about class and line number location)
39   * is emitted as well.
40   *
41   * @author <a href="mailto:kaare.nilsen@gmail.com">Kaare Nilsen</a>
42   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
43   */
44  public class MavenMessageHandler extends MessageHandler {
45  
46  
47      // Internal state
48      private static final List<IMessage.Kind> DEFAULT_DETAIL_TYPES;
49      private Log log;
50      private List<IMessage.Kind> showDetailsForMessageKindList;
51  
52      static {
53          DEFAULT_DETAIL_TYPES = new ArrayList<IMessage.Kind>();
54          DEFAULT_DETAIL_TYPES.add(IMessage.ERROR);
55          DEFAULT_DETAIL_TYPES.add(IMessage.WARNING);
56          DEFAULT_DETAIL_TYPES.add(IMessage.FAIL);
57      }
58  
59      /**
60       * Constructs a MessageHandler with a Maven plugin logger.
61       *
62       * @param log                           The active Maven Log.
63       * @param showDetailsForMessageKindList A List holding all AJC message types for which this MavenMessageHandler
64       *                                      should emit details onto the Maven log (i.e. class name,
65       *                                      line/row number etc.)
66       */
67      public MavenMessageHandler(final Log log,
68                                 final List<IMessage.Kind> showDetailsForMessageKindList) {
69  
70          // Check sanity
71          // assert log != null : "Cannot handle null log argument.";
72          // assert showDetailsForMessageKindList != null : "Cannot handle null showDetailsForMessageKindList argument.";
73          if (log == null) {
74              throw new NullPointerException("Cannot handle null log argument.");
75          }
76          if (showDetailsForMessageKindList == null) {
77              throw new NullPointerException("Cannot handle null showDetailsForMessageKindList argument.");
78          }
79  
80          // Assign internal state
81          this.log = log;
82          this.showDetailsForMessageKindList = showDetailsForMessageKindList;
83  
84          if (log.isInfoEnabled()) {
85              log.info("Showing AJC message detail for messages of types: " + showDetailsForMessageKindList);
86          }
87      }
88  
89      /**
90       * Constructs a MessageHandler with a Maven plugin logger, and emitting detailed information for all
91       * AJC message kinds the {@code DEFAULT_DETAIL_TYPES} List.
92       *
93       * @param log The active Maven Log.
94       */
95      public MavenMessageHandler(final Log log) {
96          this(log, DEFAULT_DETAIL_TYPES);
97      }
98  
99      /**
100      * Copies output from the supplied message onto the active Maven Log.
101      * If the message type (i.e. {@code message.getKind()}) is listed in the showDetailsForMessageKindList List,
102      * the message is prefixed with location details (Class, row/line number etc.) as well.
103      * <p/>
104      * {@inheritDoc}
105      */
106     public boolean handleMessage(final IMessage message) {
107 
108         // Compose the message text
109         final StringBuilder builder = new StringBuilder(message.getMessage());
110         if (isMessageDetailDesired(message)) {
111 
112             //
113             // The AJC details are typically delivered on the format [fileName]:[lineNumber]
114             // (i.e. /src/main/java/Clazz.java:16).
115             //
116             // Mimic this, and include the context of the message as well,
117             // including guarding against NPEs.
118             //
119             final ISourceLocation sourceLocation = message.getSourceLocation();
120             final String sourceFile = sourceLocation == null || sourceLocation.getSourceFile() == null
121                     ? "<unknown source file>"
122                     : sourceLocation.getSourceFile().getAbsolutePath();
123             final String context = sourceLocation == null || sourceLocation.getContext() == null
124                     ? ""
125                     : sourceLocation.getContext() + "\n";
126             final String line = sourceLocation == null
127                     ? "<no line information>"
128                     : "" + sourceLocation.getLine();
129 
130             builder.append("\n\t")
131                     .append(sourceFile)
132                     .append(":")
133                     .append(line)
134                     .append("\n")
135                     .append(context);
136         }
137 
138         final String messageText = builder.toString();
139 
140         if (isNotIgnored(message, IMessage.DEBUG)
141                 || isNotIgnored(message, IMessage.INFO)
142                 || isNotIgnored(message, IMessage.TASKTAG)) {
143 
144             // The DEBUG, INFO, and TASKTAG ajc message kinds are considered Maven Debug messages.
145             log.debug(messageText);
146 
147         } else if (isNotIgnored(message, IMessage.WEAVEINFO)) {
148 
149             // The WEAVEINFO ajc message kind is considered Maven Info messages.
150             log.info(messageText);
151 
152         } else if (isNotIgnored(message, IMessage.WARNING)) {
153 
154             // The WARNING ajc message kind is considered Maven Warn messages.
155             log.warn(messageText);
156 
157         } else if(isNotIgnored(message, IMessage.ERROR)
158                 || isNotIgnored(message, IMessage.ABORT)
159                 || isNotIgnored(message, IMessage.FAIL)) {
160 
161             // We map ERROR, ABORT, and FAIL ajc message kinds to Maven Error messages.
162             log.error(messageText);
163         }
164 
165         // Delegate to normal handling.
166         return super.handleMessage(message);
167     }
168 
169     //
170     // Private helpers
171     //
172 
173     private boolean isMessageDetailDesired(final IMessage message) {
174 
175         if (message != null) {
176             for (IMessage.Kind current : showDetailsForMessageKindList) {
177                 if (message.getKind().equals(current)) {
178                     return true;
179                 }
180             }
181         }
182 
183         return false;
184     }
185 
186     private boolean isNotIgnored(final IMessage message, final IMessage.Kind messageType) {
187         return message.getKind().equals(messageType) && !isIgnoring(messageType);
188     }
189 }
